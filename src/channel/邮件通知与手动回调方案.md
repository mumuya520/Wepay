# WePay V3 邮件通知与手动回调方案

## 一、方案概述

### 1.1 设计目标
- 订单状态变更时自动发送邮件通知商户
- 支付失败/超时订单每日汇总邮件
- 邮件中提供手动回调按钮，商户可一键触发回调
- 回调链接带签名防伪造，支持过期时间控制
- 记录邮件发送和手动回调日志，便于追溯

### 1.2 业务场景
1. **订单支付成功**：立即发送邮件通知商户
2. **订单支付失败**：每日10:00汇总发送失败订单列表
3. **订单超时未支付**：每日汇总发送超时订单列表
4. **回调失败**：邮件中提供手动回调按钮
5. **设备离线告警**：通知商户设备异常

### 1.3 技术架构

```cpp
std::string EmailService::renderTemplate(const std::string& templateName,
                                        const nlohmann::json& data) {
    // 读取模板文件
    std::string templatePath = "templates/" + templateName + ".html";
    std::ifstream file(templatePath);
    std::stringstream buffer;
    buffer << file.rdbuf();
    std::string html = buffer.str();
    
    // 替换模板变量 {{.Variable}}
    for (auto& [key, value] : data.items()) {
        std::string placeholder = "{{." + key + "}}";
        std::string valueStr = value.is_string() ? value.get<std::string>() 
                                                  : value.dump();
        
        size_t pos = 0;
        while ((pos = html.find(placeholder, pos)) != std::string::npos) {
            html.replace(pos, placeholder.length(), valueStr);
            pos += valueStr.length();
        }
    }
    
    return html;
}

bool EmailService::sendEmail(const std::string& to,
                            const std::string& subject,
                            const std::string& htmlBody) {
    CURL* curl = curl_easy_init();
    if (!curl) return false;
    
    struct curl_slist* recipients = nullptr;
    recipients = curl_slist_append(recipients, to.c_str());
    
    // 构建邮件内容
    std::ostringstream payload;
    payload << "From: " << config_.fromName << " <" << config_.fromEmail << ">\r\n";
    payload << "To: " << to << "\r\n";
    payload << "Subject: " << subject << "\r\n";
    payload << "MIME-Version: 1.0\r\n";
    payload << "Content-Type: text/html; charset=UTF-8\r\n";
    payload << "\r\n";
    payload << htmlBody;
    
    std::string payloadStr = payload.str();
    
    // 设置CURL选项
    std::string url = "smtp://" + config_.smtpHost + ":" + std::to_string(config_.smtpPort);
    curl_easy_setopt(curl, CURLOPT_URL, url.c_str());
    curl_easy_setopt(curl, CURLOPT_USERNAME, config_.username.c_str());
    curl_easy_setopt(curl, CURLOPT_PASSWORD, config_.password.c_str());
    curl_easy_setopt(curl, CURLOPT_MAIL_FROM, config_.fromEmail.c_str());
    curl_easy_setopt(curl, CURLOPT_MAIL_RCPT, recipients);
    curl_easy_setopt(curl, CURLOPT_READDATA, &payloadStr);
    
    if (config_.useSsl) {
        curl_easy_setopt(curl, CURLOPT_USE_SSL, CURLUSESSL_ALL);
    }
    
    // 执行发送
    CURLcode res = curl_easy_perform(curl);
    bool success = (res == CURLE_OK);
    
    curl_slist_free_all(recipients);
    curl_easy_cleanup(curl);
    
    return success;
}
```

### 4.3 手动回调令牌生成器
```cpp
#pragma once
#include <string>
#include <openssl/hmac.h>
#include <openssl/sha.h>

class CallbackTokenGenerator {
public:
    struct TokenData {
        std::string orderId;
        std::string merchantId;
        int64_t expireTime;  // Unix时间戳
    };
    
    // 生成回调令牌
    static std::string generateToken(const TokenData& data, const std::string& secret);
    
    // 验证回调令牌
    static bool verifyToken(const std::string& token,
                           const std::string& orderId,
                           const std::string& merchantId,
                           const std::string& secret);
    
    // 生成完整的回调URL
    static std::string generateCallbackUrl(const std::string& baseUrl,
                                          const TokenData& data,
                                          const std::string& secret);
```

```cpp
#include "CallbackTokenGenerator.h"
#include <sstream>
#include <iomanip>
#include <ctime>

std::string CallbackTokenGenerator::generateToken(const TokenData& data,
                                                  const std::string& secret) {
    // 构建待签名字符串：orderId|merchantId|expireTime
    std::ostringstream oss;
    oss << data.orderId << "|" << data.merchantId << "|" << data.expireTime;
    std::string plainText = oss.str();
    
    // HMAC-SHA256签名
    std::string signature = hmacSha256(plainText, secret);
    
    // 组合：plainText + "." + signature，然后Base64编码
    std::string combined = plainText + "." + signature;
    return base64Encode((unsigned char*)combined.c_str(), combined.length());
}

bool CallbackTokenGenerator::verifyToken(const std::string& token,
                                        const std::string& orderId,
                                        const std::string& merchantId,
                                        const std::string& secret) {
    try {
        // Base64解码
        std::string decoded = base64Decode(token);
        
        // 分割plainText和signature
        size_t dotPos = decoded.rfind('.');
        if (dotPos == std::string::npos) return false;
        
        std::string plainText = decoded.substr(0, dotPos);
        std::string signature = decoded.substr(dotPos + 1);
        
        // 解析plainText
        std::istringstream iss(plainText);
        std::string tokenOrderId, tokenMerchantId;
        int64_t expireTime;
        
        std::getline(iss, tokenOrderId, '|');
        std::getline(iss, tokenMerchantId, '|');
        iss >> expireTime;
        
        // 验证订单号和商户ID
        if (tokenOrderId != orderId || tokenMerchantId != merchantId) {
            return false;
        }
        
        // 验证过期时间
        int64_t now = std::time(nullptr);
        if (now > expireTime) {
            return false;
        }
        
        // 验证签名
        std::string expectedSignature = hmacSha256(plainText, secret);
        return signature == expectedSignature;
        
    } catch (...) {
        return false;
    }
}

std::string CallbackTokenGenerator::generateCallbackUrl(const std::string& baseUrl,
                                                       const TokenData& data,
                                                       const std::string& secret) {
    std::string token = generateToken(data, secret);
    
    std::ostringstream oss;
    oss << baseUrl << "/api/wepay/v3/callback/manual"
        << "?orderId=" << data.orderId
        << "&merchantId=" << data.merchantId
        << "&token=" << token;
    
    return oss.str();
}

std::string CallbackTokenGenerator::hmacSha256(const std::string& data,
                                              const std::string& key) {
    unsigned char hash[SHA256_DIGEST_LENGTH];
    
    HMAC(EVP_sha256(),
         key.c_str(), key.length(),
         (unsigned char*)data.c_str(), data.length(),
         hash, nullptr);
    
    std::ostringstream oss;
    for (int i = 0; i < SHA256_DIGEST_LENGTH; i++) {
        oss << std::hex << std::setw(2) << std::setfill('0') << (int)hash[i];
    }
    
    return oss.str();
}

std::string CallbackTokenGenerator::base64Encode(const unsigned char* data, size_t len) {
    static const char* base64Chars = 
        "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/";
    
    std::string result;
    int i = 0;
    unsigned char array3[3];
    unsigned char array4[4];
    
    while (len--) {
        array3[i++] = *(data++);
        if (i == 3) {
            array4[0] = (array3[0] & 0xfc) >> 2;
            array4[1] = ((array3[0] & 0x03) << 4) + ((array3[1] & 0xf0) >> 4);
            array4[2] = ((array3[1] & 0x0f) << 2) + ((array3[2] & 0xc0) >> 6);
            array4[3] = array3[2] & 0x3f;
            
            for (i = 0; i < 4; i++) {
                result += base64Chars[array4[i]];
            }
            i = 0;
        }
    }
    
    if (i) {
        for (int j = i; j < 3; j++) {
            array3[j] = '\0';
        }
        
        array4[0] = (array3[0] & 0xfc) >> 2;
        array4[1] = ((array3[0] & 0x03) << 4) + ((array3[1] & 0xf0) >> 4);
        array4[2] = ((array3[1] & 0x0f) << 2) + ((array3[2] & 0xc0) >> 6);
        
        for (int j = 0; j < i + 1; j++) {
            result += base64Chars[array4[j]];
        }
        
        while (i++ < 3) {
            result += '=';
        }
    }
    
    return result;
}
```

## 十、安全建议

1. **令牌安全**
   - 使用HMAC-SHA256签名防止伪造
   - 设置合理的过期时间（建议24小时）
   - 令牌只能使用一次，使用后立即失效

2. **邮件安全**
   - 使用SSL/TLS加密传输
   - 不在邮件中包含敏感信息（如完整银行卡号）
   - 防止邮件钓鱼攻击

3. **回调安全**
   - 记录回调触发的IP和User-Agent
   - 限制同一订单的回调频率
